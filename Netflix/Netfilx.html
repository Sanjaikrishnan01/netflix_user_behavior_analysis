viewing_columns = ['Viewing Hours', 'Titles Watched', 'Hours Watched'] # Replace with actual column names if available

available_viewing_columns = [col for col in viewing_columns if col in df.columns]

if not available_viewing_columns:
    print("Viewing history data (e.g., Viewing Hours, Titles Watched) is not available in the dataset for analysis.")
else:
    if 'Viewing Hours' in df.columns:
        user_viewing_hours = df.groupby('User ID')['Viewing Hours'].sum()
        print("\nDistribution of total viewing hours per user:")
        display(user_viewing_hours.describe())

        # Optional: Plot a histogram to visualize the distribution
        import matplotlib.pyplot as plt
        plt.figure(figsize=(10, 6))
        plt.hist(user_viewing_hours, bins=30, edgecolor='black')
        plt.title('Distribution of Total Viewing Hours per User')
        plt.xlabel('Total Viewing Hours')
        plt.ylabel('Number of Users')
        plt.grid(axis='y', alpha=0.75)
        plt.show()

        high_viewing_threshold = user_viewing_hours.quantile(0.95)
        high_viewing_users = user_viewing_hours[user_viewing_hours >= high_viewing_threshold]
        print(f"\nUsers with high viewing hours (top 5%): {len(high_viewing_users)} out of {len(user_viewing_hours)}")
        print(f"Threshold for high viewing hours: {high_viewing_threshold:.2f} hours")
        if len(high_viewing_users) > 0:
            print("The presence of users with significantly higher viewing hours than the median/average suggests potential binge-watching behavior.")
        else:
            print("No significant outliers in viewing hours were identified, suggesting less pronounced binge-watching behavior based on this metric.")

    elif 'Titles Watched' in df.columns:
        user_titles_watched = df.groupby('User ID')['Titles Watched'].sum()
        print("\nDistribution of total titles watched per user:")
        display(user_titles_watched.describe())

        import matplotlib.pyplot as plt
        plt.figure(figsize=(10, 6))
        plt.hist(user_titles_watched, bins=30, edgecolor='black')
        plt.title('Distribution of Total Titles Watched per User')
        plt.xlabel('Total Titles Watched')
        plt.ylabel('Number of Users')
        plt.grid(axis='y', alpha=0.75)
        plt.show()

        high_titles_threshold = user_titles_watched.quantile(0.95)
        high_titles_users = user_titles_watched[user_titles_watched >= high_titles_threshold]
        print(f"\nUsers with high number of titles watched (top 5%): {len(high_titles_users)} out of {len(user_titles_watched)}")
        print(f"Threshold for high number of titles watched: {high_titles_threshold:.2f} titles")
